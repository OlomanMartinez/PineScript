HMA(l) =>
    n2ma = 2 * ta.wma(close, math.round(l / 2))
    nma = ta.wma(close, l)
    diff = n2ma - nma
    sqn = math.round(math.sqrt(l))
    n2ma1 = 2 * ta.wma(close[1], math.round(l / 2))
    nma1 = ta.wma(close[1], l)
    diff1 = n2ma1 - nma1
    sqn1 = math.round(math.sqrt(l))
    h1 = ta.wma(diff, sqn)
    h2 = ta.wma(diff1, sqn)
    [h1, h2]
